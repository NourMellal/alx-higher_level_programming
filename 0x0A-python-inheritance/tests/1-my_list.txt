>>> from io import StringIO
>>> import sys
>>> MyList = __import__("1-my-list").MyList

# Test with a list of integers in descending order
>>> m = MyList()
>>> m.extend([3, 2, 1])
>>> out = StringIO()
>>> sys.stdout = out
>>> m.print_sorted()
>>> sys.stdout = sys.__stdout__
>>> print(out.getvalue())
[1, 2, 3]
